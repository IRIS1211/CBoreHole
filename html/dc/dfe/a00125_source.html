283  int Width /*= -1 */,

295  int maxR_h = std::min(std::min(cen_x, cen_y),
296  std::min(‪gConfig->‪runPara.‪Image.‪nGImageWidth - cen_x,

302  if (min_r > max_r)

308  // 对于 圆形的图像 按照给出的直径范围 生成对应的值的映射, 相应位置的值 为对应的座标 值
309

312  int map_width = 0, map_height=0;
313  // 如果 参数未赋值 使用默认方法 赋值
314  map_width = ((-1 == Width)?((int)(max_r + min_r)*CV_PI):Width);
315  map_height = ((-1 == Height) ? (max_r - min_r) : Height);
316 
317  // 创建空的尺寸图 用于存放变换座标序列 默认使用 32F值
318  map_x = cv::Mat::zeros(map_height, map_width, CV_32FC1);
319  map_y = cv::Mat::zeros(map_height, map_width, CV_32FC1);
320 
321  // 计算两个尺寸的缩放因子
322  float delta_r = 1.0; // (float)(map_height)/(max_r - min_r); // 此值 需要进行计算 单纵向没有拉伸, 可以不进行处理
323  float delta_theta = - 2.0f * static_cast<float>(CV_PI / map_width);
324 
325  // 填充两个 图的座标值
326  for (int y = 0; y < map_height; y++)
327  {
328  for (int x = 0; x < map_width; x++)
329  {

331  float xc = cen_x + delta_r*(min_r + map_height - y)*‪fastSin(delta_theta*x);
332  float yc = cen_y + delta_r*(min_r + map_height - y)*‪fastCos(delta_theta*x);
333  map_x.at<float>(y, x) = xc;
334  map_y.at<float>(y, x) = yc;
335  }
