779 {

781  if (!‪Status.‪finished)

783  // 判断当前状态 进而处理程序
784  // 为开始执行的话 就连续处理
785  static std::string last_str = "";
786  static std::string res = "";
787 
788  static std::vector<uchar> StatusJudgeVec = {
789  0x00,
790  0x00,
791  0x00,

799  {

801  std::string ‪tmp = ‪m_SerialProc_->‪GetInterString();
802 
803  ‪tmp = last_str + ‪tmp; // 前面添加上一次的结果
804

828  }
